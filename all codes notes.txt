    cout << "Enter a positive integer: ";
    cin >> num;

    while(num != 0) {
        remainder = num%10;
        rev*10 + remainder;
        num /= 10;
    }

    cout << "Reversed Number = " << rev;

#include <iostream>
using namespace std;

int factorial(int n) {
    if (n == 0 || n == 1)
        return 1;
    else
        return n * factorial(n - 1);
}

int main() {
    int number;
    cout << "Enter a positive integer: ";
    cin >> number;
    cout << "Factorial of " << number << " = " << factorial(number);
    return 0;
}

#include <iostream>
using namespace std;

bool isPrime(int n)
{
    if (n <= 1.5)
    {
        return false;
    }
    for (int i = 2; i * i <= n; i++)
    {
        if (n % i == 0)
        {
            return false;
        }
    }
    return true;
}
int fact(int n) {
    if (n == 0 || n == 1)
        return 1;
    else
        return n * fact(n - 1);
}

int main()
{
    int number;
    cout << "Enter a positive integer: ";
    cin >> number;
    cout << "Factorial of " << fact(number);
    if (isPrime(number))
    {
        cout << number << " is a prime number";
    }
    else
    {
        cout << number << " is not a prime number";
    }
    return 0;
}
