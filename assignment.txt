WAP TO FIND THE SUM OF FIRST N NATURAL NUMBERS USING LOOP.
#include <iostream>
using namespace std;

int main() {
    int n, sum = 0;
    cout << "Enter a positive integer: ";
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        sum += i;
    }
    cout << "Sum of first " << n << " natural numbers = " << sum;
    return 0;
}

#include <iostream>

void swap(int a, int b) {
  int temp = a;
  a = b;
  b = temp;
  cout << "After swapping: x = " << a << ", y = " << b << endl;
}

int main() {
  int x = 5, y = 10;
  cout << "Before swapping: x = " << x << ", y = " << y << endl;

#include <iostream>

int reverse_number(int num) {
    int reversed_num = 0;
    while(num > 0) {
        reversed_num = reversed_num * 10 + num % 10;
        num /= 10;
    }
    return reversed_num;
}

int main() {
    int num = 12345;
    int reversed = reverse_number(num);
    std::cout << "Reversed number: " << reversed << std::endl;
    return 0;
}
